module Causality.Data.Vec where

open import Data.Bool using (false; true)
open import Data.Empty using (⊥-elim)
open import Data.Nat using (_>_; s≤s; z≤n)
open import Data.Vec using (Vec; []; _∷_; _[_]=_; count)
open import Data.Vec.Relation.Unary.Any using (here; there)
open import Data.Vec.Membership.Propositional using (_∈_)
open import Relation.Binary.Definitions using (DecidableEquality)
open import Relation.Nullary using (does; no; yes)
open import Relation.Unary using (Decidable)
open import Relation.Binary.PropositionalEquality.Core using (refl)

count>0 : ∀ {a p} {A : Set a} {P : A → Set p} {n} {xs : Vec A n} {x} (P? : Decidable P)
  → P x
  → x ∈ xs
  → count P? xs > 0
count>0 {x = x} P? Px (here refl)
  with P? x
...  | yes _  = s≤s z≤n
...  | no ¬Px = ⊥-elim (¬Px Px)
count>0 P? Px (there {x = x} y∈xs)
  with P? x
...  | yes _ = s≤s z≤n
...  | no  _ = count>0 P? Px y∈xs

count≟->0 : ∀ {a} {A : Set a} {n} {xs : Vec A n} {x} (_≟_ : DecidableEquality A)
  → x ∈ xs
  → count (_≟ x) xs > 0
count≟->0 _≟_ = count>0 (_≟ _) refl
