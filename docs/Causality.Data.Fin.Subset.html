Disjoint : ∀ {n} → Subset n → Subset n → Set
Disjoint S T = Empty (S ∩ T)

Disjoint-as-≡ : ∀ {n} {S T : Subset n}
  → Disjoint S T
  → S ∩ T ≡ ∅
Disjoint-as-≡ = Empty-unique

Disjoint-comm : ∀ {n} {S T : Subset n}
  → Disjoint S T
  → Disjoint T S
Disjoint-comm {S = S} {T} disjoint rewrite ∩-comm S T = disjoint

Disjoint-∅ʳ : ∀ {n} (S : Subset n) →
  Disjoint S ∅
Disjoint-∅ʳ S Nonempty-S∩∅ rewrite ∩-zeroʳ S = ∉⊥ (proj₂ (Nonempty-S∩∅))

Disjoint-∅ˡ : ∀ {n} (S : Subset n) →
  Disjoint ∅ S
Disjoint-∅ˡ S Nonempty-∅∩S rewrite ∩-zeroˡ S = ∉⊥ (proj₂ (Nonempty-∅∩S))

∉-∩-Disjoint : ∀ {n} {S T : Subset n} {x}
  → Disjoint S T
  → x ∉ S ∩ T
∉-∩-Disjoint Disjoint-S-T x∈S∩T = Disjoint-S-T (_ , x∈S∩T)

_⊆∖_ : ∀ {n} → Subset n → Subset n → Set
S ⊆∖ T = S ⊆ (⊤ ∖ T)

⁅_⁆₂ : ∀ {n} → Fin n × Fin n → Subset n
⁅ i , j ⁆₂ = ⁅ i ⁆ ∪ ⁅ j ⁆

∣x∣≡0⇒Empty-x : ∀ {n} {x : Subset n} → ∣ x ∣ ≡ 0 → Empty x
∣x∣≡0⇒Empty-x ∣x∣≡0 (_ , there {y = inside} y∈x) = case ∣x∣≡0 of λ()
∣x∣≡0⇒Empty-x ∣x∣≡0 (_ , there {y = outside} {xs = []} ())
∣x∣≡0⇒Empty-x ∣x∣≡0 (_ , there {y = outside} {xs = inside ∷ _} _) = case ∣x∣≡0 of λ()
∣x∣≡0⇒Empty-x ∣x∣≡0 (_ , there {y = outside} {xs = outside ∷ _} (there h)) = {!!}

∣x∣≡0⇒x≡∅ : ∀ {n} {x : Subset n} → ∣ x ∣ ≡ 0 → x ≡ ∅
∣x∣≡0⇒x≡∅ = Empty-unique ∘ ∣x∣≡0⇒Empty-x
